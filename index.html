led_blink.ino - 基本的なLED点滅プログラム

/*
 * Arduino R4 Minima - LED点滅プログラム
 * 
 * このプログラムはArduino R4のピン13に接続されたLEDを
 * 1秒間隔で点灯・消灯させます
 * 
 * 接続:
 * - LED アノード(+) → ピン13 (220Ω抵抗経由)
 * - LED カソード(-) → GND
 */

#define LED_PIN 13  // LED接続用ピン

void setup() {
  // ピン13をデジタル出力モードに設定
  pinMode(LED_PIN, OUTPUT);
  
  // シリアル通信初期化 (デバッグ用)
  Serial.begin(9600);
  Serial.println("LED Blink Program Started!");
}

void loop() {
  // LED点灯
  digitalWrite(LED_PIN, HIGH);
  Serial.println("LED: ON");
  delay(1000);  // 1000ミリ秒 (1秒) 待機
  
  // LED消灯
  digitalWrite(LED_PIN, LOW);
  Serial.println("LED: OFF");
  delay(1000);  // 1000ミリ秒 (1秒) 待機
}

led_blink_custom.ino - カスタマイズ版

#define LED_PIN 13

// オン時間とオフ時間を変数で管理
int onTime = 500;   // 500ミリ秒点灯
int offTime = 1500; // 1500ミリ秒消灯

void setup() {
  pinMode(LED_PIN, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  digitalWrite(LED_PIN, HIGH);
  Serial.print("LED ON for ");
  Serial.print(onTime);
  Serial.println("ms");
  delay(onTime);
  
  digitalWrite(LED_PIN, LOW);
  Serial.print("LED OFF for ");
  Serial.print(offTime);
  Serial.println("ms");
  delay(offTime);
}
